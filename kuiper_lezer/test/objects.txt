{
    "test": 123,
    concat("test", "test"): 321,
}

==>

Program(Expression(Term(Object(CommaSep(
    ObjElem(
        Expression(Term(String)),
        Expression(Term(Number))
    ),
    ObjElem(
        Expression(Term(FunctionCall(
            FunctionName(Var(PlainVar)),
            CommaSep(
                FunctionArg(Expression(Term(String))),
                FunctionArg(Expression(Term(String)))
            )
        ))),
        Expression(Term(Number))
    )
)))))
