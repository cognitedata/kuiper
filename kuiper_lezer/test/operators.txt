# Trivial

1 + 1

==>

Result(Expression(Operator(
    Expression(Term(Number)),
    ArithOp("+"),
    Expression(Term(Number))
)))

# Simple precedence

1 * 1 + 1

==>

Result(Expression(Operator(
    Expression(Operator(
        Expression(Term(Number)),
        ArithOp("*"),
        Expression(Term(Number))
    )),
    ArithOp("+"),
    Expression(Term(Number))
)))

# Simple precedence 2

1 + 1 * 1

==>

Result(Expression(Operator(
    Expression(Term(Number)),
    ArithOp("+"),
    Expression(Operator(
        Expression(Term(Number)),
        ArithOp("*"),
        Expression(Term(Number))
    ))
)))

# Complex precedence 1

1 == 2 && 2 != 3 || 1 + 1 * 3 - 5 / 3

==>

Result(Expression(Operator(
    Expression(Operator(
        Expression(Operator(
            Expression(Term(Number)),
            CompareOp("=="),
            Expression(Term(Number))
        )),
        LogicOp("&&"),
        Expression(Operator(
            Expression(Term(Number)),
            CompareOp("!="),
            Expression(Term(Number))
        )),
    )),
    LogicOp("||"),
    Expression(Operator(
        Expression(Operator(
            Expression(Term(Number)),
            ArithOp("+"),
            Expression(Operator(
                Expression(Term(Number)),
                ArithOp("*"),
                Expression(Term(Number))
            )),
        )),
        ArithOp("-"),
        Expression(Operator(
            Expression(Term(Number)),
            ArithOp("/"),
            Expression(Term(Number))
        )),
    )),
)))

# Complex precedence 2

2 + !(1 + !3 - 5)

==>

Result(Expression(Operator(
    Expression(Term(Number)),
    ArithOp("+"),
    Expression(Operator(
        LogicOp("!"),
        Expression(Term(
            Expression(Operator(
                Expression(Operator(
                    Expression(Term(Number)),
                    ArithOp("+"),
                    Expression(Operator(
                        LogicOp("!"),
                        Expression(Term(Number))
                    ))
                )),
                ArithOp("-"),
                Expression(Term(Number))
            ))
        ))
    ))
)))

# Complex precedence 3

2 + !!!3

==>

Result(Expression(Operator(
    Expression(Term(Number)),
    ArithOp("+"),
    Expression(Operator(
        LogicOp("!"),
        Expression(Operator(
            LogicOp("!"),
            Expression(Operator(
                LogicOp("!"),
                Expression(Term(Number))
            ))
        ))
    ))
)))

# Comment in expression

1 /* This is the number 1 */ + /* This is the operator for adding two numbers */ 1 /* This is also the number 1 */

==>

Result(Expression(Operator(
    Expression(Term(Number)),
    BlockComment,
    ArithOp("+"),
    BlockComment,
    Expression(Term(Number))
)), BlockComment)

# "is" operator

1 is number && "wow" is string && null is null

==>

Result(Expression(Operator(
    Expression(Operator(
        Expression(Operator(
            Expression(Term(Number)),
            CompareOp("is"),
            Type(NumberTy)
        ))
        LogicOp("&&"),
        Expression(Operator(
            Expression(Term(String)),
            CompareOp("is"),
            Type(StringTy)
        ))
    )),
    LogicOp("&&"),
    Expression(Operator(
        Expression(Term(Null)),
        CompareOp("is"),
        Type(Null)
    ))
)))

# "is not" operator

1 is not number

==>

Result(Expression(Operator(Expression(Term(Number)), CompareOp("is"), NotModifier, Type(NumberTy))))
